-- ++(0,-#2/2) node (#3_input2) at ++(0,0){\scriptsize{\hspace{8pt}0}}

%% Bus
%% 3 params : from, to, size 
\newcommand*{\bus}[4] {
\draw[->] #1 -- #2 node[midway]{\tiny{/}} 
    node [#3] {\scriptsize{#4}}
}
